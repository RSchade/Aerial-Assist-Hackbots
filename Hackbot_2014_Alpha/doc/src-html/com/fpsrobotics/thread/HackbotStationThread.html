001package com.fpsrobotics.thread;
002
003
004import com.fpsrobotics.Dashboard;
005import com.fpsrobotics.constants.*;
006import com.fpsrobotics.hardware.*;
007
008/**
009 * Uses methods from dashboard outputs to control what gets outputted to the smart dashboard.
010 * @author ray
011 */
012public class HackbotStationThread extends Thread
013{
014
015    boolean isInterrupted = false;
016
017    /**
018     *
019     * Class which outputs most variables to the SmartDashboard and gives the
020     * state of the battery through DashboardOutput's methods.
021     *
022     */
023    public void run()
024    {   
025        long previousTime = System.currentTimeMillis();
026        isInterrupted = false;
027
028        while (!isInterrupted)
029        {
030            //Everything outputs every second, to reduce lag and heat
031            if (System.currentTimeMillis() - previousTime >= Constants.THREAD_REFRESH_RATE)
032            {
033                // Output variables to dashboard
034                ThreadsAndClasses.dashboardOutputs.teamOutput();
035                ThreadsAndClasses.dashboardOutputs.outputToDashboard(Joysticks.LEFT, Joysticks.RIGHT, DigitalIOs.LEFT_DRIVE_ENCODER, Dashboard.getDistance(), Dashboard.getSpeed(), Analogs.DISTANCE_SENSOR, Analogs.SHOOTER_POTENTIOMETER);
036
037                // Reset timer to current time
038                previousTime = System.currentTimeMillis();
039            }
040        }
041
042    }
043
044    public void interrupt()
045    {
046        isInterrupted = true;
047    }
048}
