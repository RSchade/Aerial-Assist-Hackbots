009 * @author Matthew
010 */
011public class PresetAuto extends Preset
012{
013
014    public PresetAuto()
015    {
016        addValue(new PresetValue(_50_PERCENT_THRUST, 170.0));       // Start shot
017        addValue(new PresetValue(_80_PERCENT_THRUST, 250.0));       // accelerate
018        addValue(new PresetValue(_100_PERCENT_THRUST, 350));      // Full thrust
019        addValue(new PresetValue(GO_HOME, 160.0));
020        addValue(new PresetValue(STOP_SHOOTER, 160.0));
021    }
022}
