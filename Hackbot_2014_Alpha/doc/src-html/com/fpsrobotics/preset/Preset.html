005package com.fpsrobotics.preset;
006
007import java.util.Enumeration;
008import java.util.Vector;
009
010/**
011 * preset for truss shot
012 *
013 * @author Matthew
014 */
015public abstract class Preset
016{
017    protected Vector preset = null;
018    protected final double GO_HOME = -0.2;              // 40% reverse thrust to go home
019    public final static double STOP_SHOOTER = 0.0;
020    protected final double _1_PERCENT_THRUST = 0.01;    // 1% forward thrust
021    protected final double _2_PERCENT_THRUST = 0.02;
022    protected final double _5_PERCENT_THRUST = 0.05;    // 5% forward thrust
023    protected final double _10_PERCENT_THRUST = 0.1;    // 10% forward thrust
024    protected final double _20_PERCENT_THRUST = 0.2;    // 20% forward thrust
025    protected final double _30_PERCENT_THRUST = 0.3;    // 30% forward thrust
026    protected final double _40_PERCENT_THRUST = 0.4;    // 40% forward thrust
027    protected final double _50_PERCENT_THRUST = 0.5;    // 50% forward thrust
028    protected final double _60_PERCENT_THRUST = 0.6;    // 60% forward thrust
029    protected final double _70_PERCENT_THRUST = 0.7;    // 70% forward thrust
030    protected final double _80_PERCENT_THRUST = 0.8;    // 80% forward thrust
031    protected final double _90_PERCENT_THRUST = 0.9;    // 90% forward thrust
032    protected final double _100_PERCENT_THRUST = 1.0;   // 100% forward thrust
033    
034    
035    public Preset()
036    {
037        preset = new Vector();
038    }
039    
040    protected void addValue(PresetValue value)
041    {
042       this.preset.addElement(value);
043       
044    }
045    public Enumeration getElements()
046    {
047        return preset.elements();
048    }
049    
050    
051   
052}
