008/**
009 * Sample program to use NIVision to find rectangles in the scene that are
010 * illuminated by a LED ring light (similar to the model from FIRSTChoice). The
011 * camera sensitivity is set very low so as to only show light sources and
012 * remove any distracting parts of the image.
013 *
014 * The CriteriaCollection is the set of criteria that is used to filter the set
015 * of rectangles that are detected. In this example we're looking for rectangles
016 * with a minimum width of 30 pixels and maximum of 400 pixels.
017 *
018 * The algorithm first does a color threshold operation that only takes objects
019 * in the scene that have a bright green color component. Then a small object
020 * filter removes small particles that might be caused by green reflection
021 * scattered from other parts of the scene. Finally all particles are scored on
022 * rectangularity, and aspect ratio, to determine if they are a target.
023 *
024 * Look in the VisionImages directory inside the project that is created for the
025 * sample images.
026 */

088    public boolean autoImageFind() throws AxisCameraException
089    {
090        TargetReport target = new TargetReport();
091        int verticalTargets[] = new int[MAX_PARTICLES];
092        int horizontalTargets[] = new int[MAX_PARTICLES];
093        int verticalTargetCount, horizontalTargetCount;
094
095        try
096        {
097            /**
098             * Do the image capture with the camera and apply the algorithm
099             * described above. This sample will either get images from the
100             * camera or from an image file stored in the top level directory in
101             * the flash memory on the cRIO. The file name in this case is
102             * "testImage.jpg"
103             *
104             */
105
106            ColorImage image = camera.getImage();     // comment if using stored images
107            //ColorImage image;                           // next 2 lines read image from flash on cRIO
108            //image = new RGBImage("/testImage.jpg");           // get the sample image from the cRIO flash
109            // BinaryImage thresholdImage = image.thresholdRGB(0, 255, 200, 255, 0, 255);   // keep only green objects
110            BinaryImage thresholdImage = image.thresholdHSL(0, 255, 0, 255, 245, 255); // HSV values need to be tuned
111//            thresholdImage.write("/threshold.bmp");
112            BinaryImage filteredImage = thresholdImage.particleFilter(cc);           // filter out small particles
113            filteredImage.write("/filteredImageHotGoal.bmp");
114
115            //iterate through each particle and score to see if it is a target
116            Scores scores[] = new Scores[filteredImage.getNumberParticles()];
117            horizontalTargetCount = verticalTargetCount = 0;
118
119            if (filteredImage.getNumberParticles() > 0)
120            {
121                for (int i = 0; i < MAX_PARTICLES && i < filteredImage.getNumberParticles(); i++)
122                {
123                    System.out.println(i);
124
125                    ParticleAnalysisReport report = filteredImage.getParticleAnalysisReport(i);
126
127                    System.out.println("Height " + report.boundingRectHeight);
128                    System.out.println("Width " + report.boundingRectWidth);
129
130                    scores[i] = new Scores();
131
132                    //Score each particle on rectangularity and aspect ratio
133                    scores[i].rectangularity = scoreRectangularity(report);
134                    scores[i].aspectRatioVertical = scoreAspectRatio(filteredImage, report, i, true);
135                    scores[i].aspectRatioHorizontal = scoreAspectRatio(filteredImage, report, i, false);
136
137                    //Check if the particle is a horizontal target, if not, check if it's a vertical target
138                    if (scoreCompareTarget(scores[i], false))
139                    {
140                        filteredImage.free();
141                        thresholdImage.free();
142                        image.free();
143
144                        System.out.println("particle: " + i + "is a Horizontal Target centerX: " + report.center_mass_x + "centerY: " + report.center_mass_y);
145                        horizontalTargets[horizontalTargetCount++] = i; //Add particle to target array and increment count
146                        return true;
147                    } else if (scoreCompareTarget(scores[i], true))
148                    {
149                        System.out.println("particle: " + i + "is a Vertical Target centerX: " + report.center_mass_x + "centerY: " + report.center_mass_y);
150                        verticalTargets[verticalTargetCount++] = i;  //Add particle to target array and increment count
151                        return true;
152                    } else
153                    {
154                        System.out.println("particle: " + i + "is not a Target centerX: " + report.center_mass_x + "centerY: " + report.center_mass_y);
155
156                    }
157                    System.out.println("rect: " + scores[i].rectangularity + "ARHoriz: " + scores[i].aspectRatioHorizontal);
158                    System.out.println("ARVert: " + scores[i].aspectRatioVertical);
159
160                }
161
162                //Zero out scores and set verticalIndex to first target in case there are no horizontal targets
163                target.totalScore = target.leftScore = target.rightScore = target.tapeWidthScore = target.verticalScore = 0;
164                target.verticalIndex = verticalTargets[0];
165                for (int i = 0; i < verticalTargetCount; i++)
166                {
167                    ParticleAnalysisReport verticalReport = filteredImage.getParticleAnalysisReport(verticalTargets[i]);
168                    for (int j = 0; j < horizontalTargetCount; j++)
169                    {
170                        ParticleAnalysisReport horizontalReport = filteredImage.getParticleAnalysisReport(horizontalTargets[j]);
171                        double horizWidth, horizHeight, vertWidth, leftScore, rightScore, tapeWidthScore, verticalScore, total;
172
173                        //Measure equivalent rectangle sides for use in score calculation
174                        horizWidth = NIVision.MeasureParticle(filteredImage.image, horizontalTargets[j], false, MeasurementType.IMAQ_MT_EQUIVALENT_RECT_LONG_SIDE);
175                        vertWidth = NIVision.MeasureParticle(filteredImage.image, verticalTargets[i], false, MeasurementType.IMAQ_MT_EQUIVALENT_RECT_SHORT_SIDE);
176                        horizHeight = NIVision.MeasureParticle(filteredImage.image, horizontalTargets[j], false, MeasurementType.IMAQ_MT_EQUIVALENT_RECT_SHORT_SIDE);
177
178                        //Determine if the horizontal target is in the expected location to the left of the vertical target
179                        leftScore = ratioToScore(1.2 * (verticalReport.boundingRectLeft - horizontalReport.center_mass_x) / horizWidth);
180                        //Determine if the horizontal target is in the expected location to the right of the  vertical target
181                        rightScore = ratioToScore(1.2 * (horizontalReport.center_mass_x - verticalReport.boundingRectLeft - verticalReport.boundingRectWidth) / horizWidth);
182                        //Determine if the width of the tape on the two targets appears to be the same
183                        tapeWidthScore = ratioToScore(vertWidth / horizHeight);
184                        //Determine if the vertical location of the horizontal target appears to be correct
185                        verticalScore = ratioToScore(1 - (verticalReport.boundingRectTop - horizontalReport.center_mass_y) / (4 * horizHeight));
186                        total = leftScore > rightScore ? leftScore : rightScore;
187                        total += tapeWidthScore + verticalScore;
188
189                        //If the target is the best detected so far store the information about it
190                        if (total > target.totalScore)
191                        {
192                            target.horizontalIndex = horizontalTargets[j];
193                            target.verticalIndex = verticalTargets[i];
194                            target.totalScore = total;
195                            target.leftScore = leftScore;
196                            target.rightScore = rightScore;
197                            target.tapeWidthScore = tapeWidthScore;
198                            target.verticalScore = verticalScore;
199                        }
200                    }
201                    //Determine if the best target is a Hot target
202                    target.Hot = hotOrNot(target);
203                }
204
205                if (verticalTargetCount > 0)
206                {
207                    //Information about the target is contained in the "target" structure
208                    //To get measurement information such as sizes or locations use the
209                    //horizontal or vertical index to get the particle report as shown below
210                    ParticleAnalysisReport distanceReport = filteredImage.getParticleAnalysisReport(target.verticalIndex);
211                    double distance = computeDistance(filteredImage, distanceReport, target.verticalIndex);
212                    if (target.Hot)
213                    {
214                        System.out.println("Hot target located");
215                        System.out.println("Distance: " + distance);
216                    } else
217                    {
218                        System.out.println("No hot target present");
219                        System.out.println("Distance: " + distance);
220                    }
221                }
222
223            }
224
225            /**
226             * all images in Java must be freed after they are used since they
227             * are allocated out of C data structures. Not calling free() will
228             * cause the memory to accumulate over each pass of this loop.
229             */
230            filteredImage.free();
231            thresholdImage.free();
232            image.free();
233
234        } catch (AxisCameraException ex)
235        {        // this is needed if the camera.getImage() is called
236            ex.printStackTrace();
237        } catch (NIVisionException ex)
238        {
239            ex.printStackTrace();
240        }
241
242        return false;
243
244    }

326    /**
327     * Computes the estimated distance to a target using the height of the
328     * particle in the image. For more information and graphics showing the math
329     * behind this approach see the Vision Processing section of the
330     * ScreenStepsLive documentation.
331     *
332     * @param image The image to use for measuring the particle estimated
333     * rectangle
334     * @param report The Particle Analysis Report for the particle
335     * @param outer True if the particle should be treated as an outer target,
336     * false to treat it as a center target
337     * @return The estimated distance to the target in Inches.
338     */
339    double computeDistance(BinaryImage image, ParticleAnalysisReport report, int particleNumber) throws NIVisionException
340    {
341        double rectLong, height;
342        int targetHeight;
343
344        rectLong = NIVision.MeasureParticle(image.image, particleNumber, false, MeasurementType.IMAQ_MT_EQUIVALENT_RECT_LONG_SIDE);
345        //using the smaller of the estimated rectangle long side and the bounding rectangle height results in better performance
346        //on skewed rectangles
347        height = Math.min(report.boundingRectHeight, rectLong);
348        targetHeight = 32;
349
350        return Y_IMAGE_RES * targetHeight / (height * 12 * 2 * Math.tan(VIEW_ANGLE * Math.PI / (180 * 2)));
351    }

353    /**
354     * Computes a score (0-100) comparing the aspect ratio to the ideal aspect
355     * ratio for the target. This method uses the equivalent rectangle sides to
356     * determine aspect ratio as it performs better as the target gets skewed by
357     * moving to the left or right. The equivalent rectangle is the rectangle
358     * with sides x and y where particle area= x*y and particle perimeter= 2x+2y
359     *
360     * @param image The image containing the particle to score, needed to
361     * perform additional measurements
362     * @param report The Particle Analysis Report for the particle, used for the
363     * width, height, and particle number
364     * @param outer     Indicates whether the particle aspect ratio should be
365     * compared to the ratio for the inner target or the outer
366     * @return The aspect ratio score (0-100)
367     */
368    public double scoreAspectRatio(BinaryImage image, ParticleAnalysisReport report, int particleNumber, boolean vertical) throws NIVisionException
369    {
370        double rectLong, rectShort, aspectRatio, idealAspectRatio;
371
372        rectLong = NIVision.MeasureParticle(image.image, particleNumber, false, MeasurementType.IMAQ_MT_EQUIVALENT_RECT_LONG_SIDE);
373        rectShort = NIVision.MeasureParticle(image.image, particleNumber, false, MeasurementType.IMAQ_MT_EQUIVALENT_RECT_SHORT_SIDE);
374        idealAspectRatio = vertical ? (4.0 / 32) : (23.5 / 4);  //Vertical reflector 4" wide x 32" tall, horizontal 23.5" wide x 4" tall
375
376        //Divide width by height to measure aspect ratio
377        if (report.boundingRectWidth > report.boundingRectHeight)
378        {
379            //particle is wider than it is tall, divide long by short
380            aspectRatio = ratioToScore((rectLong / rectShort) / idealAspectRatio);
381        } else
382        {
383            //particle is taller than it is wide, divide short by long
384            aspectRatio = ratioToScore((rectShort / rectLong) / idealAspectRatio);
385        }
386        return aspectRatio;
387    }

389    /**
390     * Compares scores to defined limits and returns true if the particle
391     * appears to be a target
392     *
393     * @param scores The structure containing the scores to compare
394     * @param outer True if the particle should be treated as an outer target,
395     * false to treat it as a center target
396     *
397     * @return True if the particle meets all limits, false otherwise
398     */
399    boolean scoreCompareTarget(Scores scores, boolean vertical)
400    {
401        boolean isTarget = true;
402
403        isTarget &= scores.rectangularity > RECTANGULARITY_LIMIT;
404        if (vertical)
405        {
406            isTarget &= scores.aspectRatioVertical > ASPECT_RATIO_LIMIT_TARGET;
407        } else
408        {
409            isTarget &= scores.aspectRatioHorizontal > ASPECT_RATIO_LIMIT_TARGET;
410        }
411
412        return isTarget;
413    }
414
415    boolean scoreCompareBumper(ScoresBumper scores)
416    {
417        boolean isTarget = true;
418
419        isTarget &= scores.rectangularity > RECTANGULARITY_LIMIT;
420        isTarget &= scores.aspectRatioHorizontal > ASPECT_RATIO_LIMIT_BUMPER;
421
422        return isTarget;
423    }

425    /**
426     * Computes a score (0-100) estimating how rectangular the particle is by
427     * comparing the area of the particle to the area of the bounding box
428     * surrounding it. A perfect rectangle would cover the entire bounding box.
429     *
430     * @param report The Particle Analysis Report for the particle to score
431     * @return The rectangularity score (0-100)
432     */
433    double scoreRectangularity(ParticleAnalysisReport report)
434    {
435        if (report.boundingRectWidth * report.boundingRectHeight != 0)
436        {
437            return 100 * report.particleArea / (report.boundingRectWidth * report.boundingRectHeight);
438        } else
439        {
440            return 0;
441        }
442    }
443
444    /**
445     * Converts a ratio with ideal value of 1 to a score. The resulting function
446     * is piecewise linear going from (0,0) to (1,100) to (2,0) and is 0 for all
447     * inputs outside the range 0-2
448     */
449    double ratioToScore(double ratio)
450    {
451        return (Math.max(0, Math.min(100 * (1 - Math.abs(1 - ratio)), 100)));
452    }

454    /**
455     * Takes in a report on a target and compares the scores to the defined
456     * score limits to evaluate if the target is a hot target or not.
457     *
458     * Returns True if the target is hot. False if it is not.
459     */
460    boolean hotOrNot(TargetReport target)
461    {
462        boolean isHot = true;
463
464        isHot &= target.tapeWidthScore >= TAPE_WIDTH_LIMIT;
465        isHot &= target.verticalScore >= VERTICAL_SCORE_LIMIT;
466        isHot &= (target.leftScore > LR_SCORE_LIMIT) | (target.rightScore > LR_SCORE_LIMIT);
467
468        return isHot;
469    }
