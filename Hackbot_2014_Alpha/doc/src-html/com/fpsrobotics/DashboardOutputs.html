021    /**
022     * Outputs values to SmartDashboard.
023     *
024     * @param leftJoystick
025     * @param rightJoystick
026     * @param encoder
027     * @param dynamicDistance
028     * @param dynamicSpeed
029     * @param distanceSensor
030     * @param shooterPot
031     */
032    public void outputToDashboard(Joystick leftJoystick, Joystick rightJoystick, Encoder encoder, double dynamicDistance, double dynamicSpeed, AnalogChannel distanceSensor, AnalogChannel shooterPot)
033    {
034        // Variable outputs to dashboard
035        SmartDashboard.putNumber("Right Drive Train Speed", rightJoystick.getRawAxis(2));
036        SmartDashboard.putNumber("Left Drive Train Speed", leftJoystick.getRawAxis(2));
037        SmartDashboard.putNumber("Threads Currently Running", Thread.activeCount());
038        SmartDashboard.putNumber("Encoder Rate", encoder.getRate());
039        SmartDashboard.putNumber("Encoder Value", encoder.getRaw());
040        SmartDashboard.putNumber("Dynamic Preset Distance", dynamicDistance);
041        SmartDashboard.putNumber("Dynamic Preset Speed", dynamicSpeed);
042        SmartDashboard.putNumber("Ultrasonic Distance (ft)", distanceSensor.getVoltage()/.1176);
043        SmartDashboard.putNumber("Shooter Pot", shooterPot.getValue());
044    }
